class Player:
    def __init__(self, name):
        self.name = name
        self.location = "kasernen"
        self.inventory = []
        self.eggs_found = []
        self.eggs_decorated = []

    def move_to(self, destination):
        loc = locations[self.location]
        if destination in loc["udgange"]:
            self.location = destination
            return True
        return False

    def pick_up(self, item):
        self.inventory.append(item)

    def has_item(self, item):
        return item in self.inventory

    def found_egg(self, egg):
        if egg not in self.eggs_found:
            self.inventory.append(egg)
            self.eggs_found.append(egg)
            return True
        return False

    def decorate_egg(self, egg, decoration):
        if egg in self.inventory and decoration in self.inventory and egg not in self.eggs_decorated:
            self.eggs_decorated.append(egg)
            self.inventory.remove(decoration)
            return True
        return False

    def print_status(self):
        print(f"👤 {self.name} er på {locations[self.location]['navn']}")
        print(f"🥚 Æg fundet: {len(self.eggs_found)}/4")
        print(f"🎨 Dekorerede: {len(self.eggs_decorated)}/4")

# Brug klassen i spillet:
